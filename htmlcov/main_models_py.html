27 def send_verification_email(self): 

28 payload = {'email': self.email, 'code': self.get_confirmation_key()} 

29 result = urlencode(payload, quote_via=quote_plus) 

30 d = {'result': result, 'site_url': settings.SITE_URL} 

31 subject, from_email, to = "Email verification", "no-reply@example.com", self.email 

32 text_content = get_template("main/email/email-confirmation.txt").render(d) 

33 html_content = get_template("main/email/email-confirmation.html").render(d) 

34 msg = EmailMultiAlternatives(subject, text_content, from_email, [to]) 

35 msg.attach_alternative(html_content, "text/html") 

36 msg.send()
